```javascript
// Sistema de tabs
function inicializarTabs() {
    const botoesTabs = document.querySelectorAll('[data-action="abrir-tab"]');
    
    botoesTabs.forEach(btn => {
        btn.addEventListener('click', () => {
            const targetId = btn.dataset.tabTarget;
            
            // Remove active de todos
            document.querySelectorAll('.tabs__btn').forEach(b => b.classList.remove('tabs__btn--active'));
            document.querySelectorAll('.tabs__content').forEach(c => c.classList.remove('tabs__content--active'));
            
            // Adiciona active no clicado
            btn.classList.add('tabs__btn--active');
            document.getElementById(targetId)?.classList.add('tabs__content--active');
            
            // Limpar busca
            const searchInput = document.getElementById('searchInput');
            if (searchInput) {
                searchInput.value = '';
                filtrarRamais();
            }
        });
    });
}

// Busca de ramais
function filtrarRamais() {
    const input = document.getElementById('searchInput');
    const termo = normalizeText(input?.value || '');
    
    const abaAtiva = document.querySelector('.tabs__content--active');
    if (!abaAtiva) return;
    
    let totalVisiveis = 0;
    const cards = abaAtiva.querySelectorAll('.ramal-card');
    
    cards.forEach(card => {
        const searchAttr = card.dataset.search || '';
        const textoCard = card.textContent || '';
        const textoCompleto = normalizeText(searchAttr + ' ' + textoCard);
        
        if (textoCompleto.includes(termo)) {
            card.style.display = '';
            totalVisiveis++;
        } else {
            card.style.display = 'none';
        }
    });
    
    // Ocultar subgrupos vazios
    const subgrupos = abaAtiva.querySelectorAll('.ramais-subgroup');
    subgrupos.forEach(grupo => {
        const cardsVisiveis = Array.from(grupo.querySelectorAll('.ramal-card'))
            .filter(c => c.style.display !== 'none');
        grupo.style.display = cardsVisiveis.length > 0 ? '' : 'none';
    });
    
    // Atualizar contador
    const contador = abaAtiva.querySelector('.ramais-group__counter');
    if (contador) {
        if (termo === '') {
            contador.textContent = '';
        } else if (totalVisiveis === 0) {
            contador.textContent = 'Nenhum contato encontrado.';
        } else {
            contador.textContent = `${totalVisiveis} contato(s) encontrado(s)`;
        }
    }
}

function normalizeText(text) {
    return text
        .normalize('NFD')
        .replace(/[\u0300-\u036f]/g, '')
        .toLowerCase();
}

// Inicializar
document.addEventListener('DOMContentLoaded', () => {
    inicializarTabs();
    
    const searchInput = document.getElementById('searchInput');
    searchInput?.addEventListener('input', filtrarRamais);
    
    // Iniciar busca vazia para configurar contadores
    filtrarRamais();
});
```

```javascript
// Carregador de tutoriais via fetch
async function carregarTutorial(slug) {
    const area = document.getElementById('tutorialsContent');
    if (!area) return;
    
    area.innerHTML = '<p class="tutorials-content__loading">Carregando tutorial...</p>';
    
    try {
        const resposta = await fetch(`tutoriais/${slug}.html`);
        
        if (!resposta.ok) {
            area.innerHTML = '<p class="tutorials-content__error">Não foi possível carregar este tutorial.</p>';
            return;
        }
        
        area.innerHTML = await resposta.text();
        
        // Inicializar componentes do tutorial carregado
        inicializarTutorial(area);
        
    } catch (erro) {
        console.error('Erro ao carregar tutorial:', erro);
        area.innerHTML = '<p class="tutorials-content__error">Erro ao carregar o tutorial. Tente novamente.</p>';
    }
}

function inicializarTutorial(container) {
    // Tabs internas do tutorial
    const badgesTabs = container.querySelectorAll('[data-action="trocar-tab"]');
    badgesTabs.forEach(badge => {
        badge.addEventListener('click', () => {
            const targetId = badge.dataset.tabTarget;
            
            // Remove active de badges
            const pai = badge.closest('.tutorial-tabs');
            pai?.querySelectorAll('[data-action="trocar-tab"]').forEach(b => b.classList.remove('badge--active'));
            badge.classList.add('badge--active');
            
            // Mostra conteúdo
            const tutorial = badge.closest('.tutorial-box');
            tutorial?.querySelectorAll('.tutorial-tab-content').forEach(c => c.style.display = 'none');
            tutorial?.querySelector(`#${targetId}`)?.style.display = 'block';
        });
    });
    
    // Zoom de imagens do tutorial
    container.querySelectorAll('[data-action="abrir-zoom"]').forEach(img => {
        img.addEventListener('click', () => {
            const modalZoom = document.getElementById('modalZoom');
            const imgZoom = document.getElementById('imgZoom');
            if (modalZoom && imgZoom) {
                imgZoom.src = img.src;
                imgZoom.alt = img.alt;
                modalZoom.style.display = 'flex';
            }
        });
    });
}

// Navegação lateral
function inicializarNavegacaoTutoriais() {
    const linksManuais = document.querySelectorAll('[data-action="abrir-manual"]');
    
    linksManuais.forEach(link => {
        link.addEventListener('click', (e) => {
            e.preventDefault();
            
            const slug = link.dataset.manualTarget;
            
            // Remove active de todos
            document.querySelectorAll('.tutorials-nav__link').forEach(l => l.classList.remove('tutorials-nav__link--active'));
            link.classList.add('tutorials-nav__link--active');
            
            // Carregar tutorial
            carregarTutorial(slug);
        });
    });
    
    // Carregar primeiro tutorial automaticamente
    const primeiroLink = document.querySelector('[data-action="abrir-manual"]');
    if (primeiroLink) {
        const slug = primeiroLink.dataset.manualTarget;
        carregarTutorial(slug);
    }
}

// Atalho dos cards iniciais
function inicializarAtalhoCards() {
    document.querySelectorAll('[data-action="acessar-manual"]').forEach(card => {
        card.addEventListener('click', (e) => {
            e.preventDefault();
            
            const slug = card.dataset.manualTarget;
            const linkAlvo = document.querySelector(`[data-manual-target="${slug}"]`);
            
            if (linkAlvo) {
                linkAlvo.click();
                
                // Scroll para tutoriais
                setTimeout(() => {
                    const secaoTutoriais = document.getElementById('tutoriais');
                    if (secaoTutoriais) {
                        const headerHeight = document.querySelector('.header-principal')?.offsetHeight || 0;
                        const targetPosition = secaoTutoriais.getBoundingClientRect().top + window.scrollY - headerHeight - 16;
                        window.scrollTo({ top: targetPosition, behavior: 'smooth' });
                    }
                }, 100);
            }
        });
    });
}

// Inicializar
document.addEventListener('DOMContentLoaded', () => {
    inicializarNavegacaoTutoriais();
    inicializarAtalhoCards();
});
```
